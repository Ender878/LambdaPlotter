[options]

# --- Force GLFW to use the correct desktop Linux backend ---
# 1. Disable Wayland support to prevent it from being auto-detected
glfw/*:with_wayland=False
# 2. Disable EGL support, which is often associated with OpenGL ES
glfw/*:use_egl=False
# 3. Ensure GLX support (the X11 OpenGL interface) is enabled
glfw/*:use_glx=True
